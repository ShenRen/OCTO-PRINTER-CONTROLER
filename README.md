#!/bin/bash

Daemon=mjpg_streamer

DaemonBase=/usr/local

DaemonArgs="-i \"input_uvc.so –f 2 -y\" -o \"output_http.so\""

case "$1" in

start)

eval LD_LIBRARY_PATH=${DaemonBase}/lib ${DaemonBase}/bin/${Daemon} ${DaemonArgs} >/dev/null 2>&1 &

echo "$0: started"

;;
stop)

pkill -x ${Daemon}

echo "$0: stopped"

;;

*)

echo "Usage: $0 {start|stop}" >&2

;;
